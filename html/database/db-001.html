关于mybatis 3.0 collection用法
  • 用法一：作为主表的子表数据使用
//主表对象
public class EducationClass {
    private Integer id;
    private List<EducationClassManage> classManageList;
}
mybatis mapper.xml配置
<resultMap id="BaseResultMap" type="com.wxadt.wxcep.model.educationClass.EducationClass">
    <id column="id" jdbcType="INTEGER" property="id" />
    <collection property="classManageList" ofType="com.wxadt.wxcep.model.educationClass.EducationClassManage">
        <result column="classId" property="classId" />
    </collection>
</resultMap>
<select id="selectBySelective" parameterType="EducationClassForm" resultMap="BaseResultMap">
    select oec.*, group_concat(oecm.userName) as classManageNames
    from oa_education_class oec
    left join oa_education_class_manage oecm on oec.id=oecm.classId
</select>

Java对象
public class AssessmentItemPageModel extends AssessmentItem {
    private List<AssessmentItemPageModel> nodes;
}
配置用法1：
配置的column="id"会将当前记录的id字段传入select="selectItemTreeChild"这个查询中，得到的结果集将存入nodes
<resultMap type="AssessmentItemPageModel" id="TreeResultMap">
    <id column="id" jdbcType="INTEGER" property="id" />
    <result column="context" jdbcType="VARCHAR" property="text" />
    <result column="parentId" jdbcType="INTEGER" property="parentId" />
    <result column="level" jdbcType="VARCHAR" property="level" />
    <collection property="nodes" column="id" ofType="AssessmentItemPageModel" select="selectItemTreeChild"></collection>
</resultMap>

<select id="selectItemTree" parameterType="AssessmentItemForm" resultMap="TreeResultMap">
    select * from assessment_item
    where 1 = 1
    <if test="level != null">
        and level = #{level}
    </if>
    <if test="projectId != null">
        and projectId = #{projectId}
    </if>
    <if test="id != null">
        and id = #{id}
    </if>
</select>

<select id="selectItemTreeChild" parameterType="java.lang.Integer" resultType="AssessmentItemPageModel">
    select
    id, context as text, parentId, level
    from assessment_item where parentId = #{id}
</select>

配置用法2：
配置column="{id=id, schoolId=schoolId}"中的id=id。等于号前面的id是传入子节点的别名，后面的Id是当前数据的字段名。
作为子集数据parameterType的类型只能是map类型
<resultMap type="com.wxadt.wxcep.pageModel.assessment.AssessmentItemPageModel" id="ItemChildResultMap">
    <id column="id" jdbcType="INTEGER" property="id" />
    <result column="schoolId" jdbcType="INTEGER" property="schoolId" />
    <collection property="nodes" column="{id=id, schoolId=schoolId}" ofType="com.wxadt.wxcep.pageModel.assessment.AssessmentItemPageModel" select="selectItemChild"></collection>
</resultMap>

<select id="selectSchoolProject" resultMap="ItemChildResultMap">
    SELECT ai.*, aps.schoolId
    FROM assessment_item ai
    INNER JOIN assessment_project_school aps ON ai.projectId = aps.projectId
    WHERE ai.level = 1
    AND ai.projectId = #{projectId}
    AND aps.schoolId = #{schoolId}
</select>

<select id="selectItemChild" parameterType="java.util.Map" resultMap="ItemChildResultMap">
    SELECT ai.*, aps.schoolId, ais.id AS itemScoreId, ais.schoolScore, ais.areaScore, ais.status
    FROM assessment_item ai
    INNER JOIN assessment_project_school aps ON ai.projectId = aps.projectId
    LEFT JOIN assessment_item_score ais ON ai.id = ais.itemId AND aps.schoolId = ais.schoolId
    WHERE parentId = #{id}
    AND aps.schoolId = #{schoolId}
</select>
